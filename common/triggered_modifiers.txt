# Triggered modifiers are here.
# these are checked for each countries once/month and then applied.
#
# Effects are fully scriptable here.
no_infamy = {
	trigger = {
		has_global_flag = removed_infamy
	}
	
	badboy = -999
	icon = 1
}
# Vassal effects
vassal_nation = {
    trigger = {
        exists = yes
        is_vassal = yes
        is_substate = yes
    }
    icon = 1
    prestige = -0.3
}
small_puppet_master = {
    trigger = {
		num_of_vassals = 3
        NOT = { num_of_vassals = 6 }
	}
    icon = 1
    prestige = 0.1
    diplomatic_points_modifier = 0.1
}
medium_puppet_master = {
    trigger = {
		num_of_vassals = 6

	}
    icon = 1
