# Bilbo's 111th Birthday
country_event = {
	id = 50200
	picture = birthday
	title = "Bilbo's 111th Birthday"
	desc = "EVTDESC50200"
	
	is_triggered_only = yes
	
	option = {
		name = "A party of special magnificence!"
		
		prestige = 5
		
		# Gives the One Ring to Frodo if Bilbo has it
		random_country = {
			limit = {
				tag = FOR
				check_variable = {
					which = ring_controller
					value = 0 # Bilbo
				}
				NOT = {
					check_variable = {
						which = ring_controller
						value = 1 # Anyone else
					}
				}
			}
			
			set_variable = {
				which = ring_controller
				value = 1 # Frodo
			}
		}
	}
}
