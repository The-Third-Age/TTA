# Democratic Rebel Victory
country_event = {
	id = 3000
	picture = administration
	title = "Democratic Rebel Victory"
	desc = "EVTDESC3000"
	
	is_triggered_only = yes
	
	option = {
		name = "Damn"
		
		random_owned = {
			limit = {
				owner = {
					OR = {
						government = democracy
						government = ent_realm
					}
					vote_franschise = universal_weighted_voting
				}
			}
			owner = { political_reform = universal_voting }
		}
		random_owned = {
			limit = {
				owner = {
					OR = {
						government = democracy
						government = ent_realm
					}
					vote_franschise = wealth_voting
				}
			}
			owner = { political_reform = universal_weighted_voting }
		}
		
		random_owned = {
			limit = {
				owner = {
					OR = {
						government = democracy
						government = ent_realm
					}
					vote_franschise = wealth_weighted_voting
				}
			}
			owner = { political_reform = wealth_voting }
		}
		
		random_owned = {
			limit = {
				owner = {
					OR = {
						government = democracy
						government = ent_realm
					}
					vote_franschise = landed_voting
				}
			}
			owner = { political_reform = wealth_weighted_voting }
		}
		random_owned = {
			limit = {
				owner = {
					is_culture_group = ents
					NOT = { government = ent_realm }
				}
			}
			owner = {
				government = ent_realm
				political_reform = landed_voting
				set_country_flag = democratic
				election = yes
			}
		}
		random_owned = {
			limit = {
				owner = {
					NOT = { is_culture_group = ents }
					NOT = { government = democracy }
				}
			}
			owner = {
				government = democracy
				political_reform = landed_voting
				set_country_flag = democratic
				election = yes
			}
		}
	}
}

# Good Rebel win
country_event = {
	id = 3003
	picture = administration
	title = "Good Rebel Victory"
	desc = "EVTDESC3003"
	
	is_triggered_only = yes
	
	option = {
		name = "Damn"
		
		political_reform = no_slavery
		clr_country_flag = has_all_slavery
		
		random_owned = {
			limit = {
				owner = { accepted_culture = neutral_culture }
			}
			remove_accepted_culture = neutral_culture
		}
		random_owned = {
			limit = {
				owner = { accepted_culture = evil_culture }
			}
			remove_accepted_culture = evil_culture
		}
		
		political_reform = good_alignment
		add_accepted_culture = good_culture
		
		any_pop = { dominant_issue = { value = good_alignment factor = 0.2 } }
	}
}
