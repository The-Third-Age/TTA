# The Druedain Rebellion
country_event = {
	id = 150002
	picture = druedain_rebellion
	title = "The Drúedain Rebellion"
	desc = "EVTDESC150002"
	fire_only_once = yes
	
	trigger = {
		PUK = { exists = no }
		has_global_flag = annexed_druwaith_iaur
		any_pop = {
			has_pop_culture = druedain
			militancy = 4
			location = {
				is_core = PUK
				NOT = { is_core = GON }
				NOT = { is_core = ADR }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.5
			any_pop = {
				has_pop_culture = druedain
				militancy = 7
				location = {
					is_core = PUK
					NOT = { is_core = GON }
					NOT = { is_core = ADR }
				}
			}
		}
	}
	
	option = {
		name = "They wouldn't!"
		
		prestige = -5
		
		any_owned = {
			limit = {
				is_core = PUK
				NOT = { is_core = GON }
				NOT = { is_core = ADR }
				is_accepted_culture = no
				is_primary_culture = no
			}
			
			secede_province = PUK
		}
		
		PUK = {
			any_pop = {
				limit = { 
					type = labourers 
					has_pop_culture = druedain
				}
				pop_type = soldiers
			}
			country_event = { id = 150003 days = 90 }
		}
		
	}
}

# Druedain Rebellion Start
country_event = {
	id = 150003
	picture = druedain_rebellion
	title = "The Drúedain Rebellion"
	desc = "EVTDESC150003"
	
	is_triggered_only = yes
	
	option = {
		name = "To Arms!"
		
		capital_scope = {
			sub_unit = { type = archers value = current }
			sub_unit = { type = archers value = current }
			sub_unit = { type = archers value = current }
			sub_unit = { type = archers value = current }
			sub_unit = { type = pikemen value = current }
			sub_unit = { type = pikemen value = current }
			sub_unit = { type = pikemen value = current }
			sub_unit = { type = pikemen value = current }
			sub_unit = { type = scouts value = current }
			sub_unit = { type = artillery value = current }
		}
		
		treasury = 100000
			
		add_casus_belli = {
			target = FROM
			type = conquest
			months = 12
		}
		FROM = {
			war = {
				target = PUK
				attacker_goal = { casus_belli = conquest }
				defender_goal = { casus_belli = status_quo }
				call_ally = yes
			}
			set_country_flag = PUK_rebelling_from
		}
		set_global_flag = PUK_rebellion
	}
}

# Druadan Forest Rebellion
country_event = {
	id = 150007
	picture = druadan_forest_rebellion
	title = "The Drúadan Rebellion"
	desc = "EVTDESC150007"
	
	trigger = {
		owns = 11
		NOT = { primary_culture = druedain }
		NOT = { accepted_culture = druedain }
		11 = {
			average_militancy = 6
			is_core = DRU
			is_primary_culture = DRU
		}
		NOT = { exists = DRU }
		NOT = { has_global_flag = DRU_rebellion }
	}
	
	mean_time_to_happen = {
		months = 60
		
		modifier = {
			factor = 0.9
			11 = { average_consciousness = 5 }
		}
		modifier = {
			factor = 0.9
			11 = { average_consciousness = 8 }
		}
		modifier = {
			factor = 0.9
			11 = { average_militancy = 7 }
		}
		modifier = {
			factor = 0.9
			11 = { average_militancy = 8 }
		}
		modifier = {
			factor = 0.9
			11 = { average_militancy = 9 }
		}
		modifier = {
			factor = 0.5
			11 = { average_militancy = 10 }
		}
	}
	
	option = {
		name = "They wouldn't!"
		
		prestige = -3
		
		any_pop = {
			limit = { 
				type = labourers 
				has_pop_culture = druedain
				location = { province_id = 11 }
			}
			pop_type = soldiers
		}
		11 = { secede_province = DRU }
		any_pop = {
			limit = {
				has_pop_culture = druedain
				NOT = { location = { province_id = 11 } }
			}
			militancy = 1
			consciousness = 4
		}
		DRU = { country_event = { id = 150008 days = 90 } }
	}
}
