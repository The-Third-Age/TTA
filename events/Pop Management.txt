# Miltancy Menu
country_event = {
	id = 9501
	title = "Pop Militancy"
	desc = "EVTDESC9501"
	picture = administration
	
	is_triggered_only = yes
	
	option = {
		name = "Make em mad"
		
		any_pop = {
			militancy = 10
		}
	}
	
	option = {
		name = "Make unaccepteds mad"
		
		any_pop = {
			limit = {
				is_accepted_culture = no
				is_primary_culture = no
			}
			militancy = 10
		}
	}
	
	option = {
		name = "Make non-colonial unaccepteds mad"
		
		any_pop = {
			limit = {
				is_accepted_culture = no
				is_primary_culture = no
				location = { is_colonial = no }
			}
			militancy = 10
		}
	}
	
	option = {
		name = "Make accepteds happy"
		
		any_pop = {
			limit = {
				OR = {
					is_accepted_culture = yes
					is_primary_culture = yes
				}
			}
			militancy = -10
		}
	}
	
	option = {
		name = "Make em happy"
		
		any_pop = {
			militancy = -10
		}
	}
	
	option = {
		name = "Go Back"
		
		country_event = 9500
	}
	
	option = {
		name = "Quit"
	}
}

# Demotion Menu
country_event = {
	id = 9502
	title = "Pop Types"
	desc = "EVTDESC9502"
	picture = administration
	
	is_triggered_only = yes
	
	option = {
		name = "Convert all non-accepteds to labourers"
		
		any_pop = {
			limit = {
				NOT = { is_accepted_culture = yes }
				NOT = { is_primary_culture = yes }
				NOT = { type = labourers }
				NOT = { type = slaves }
			}
			pop_type = labourers
		}
	}
	
	option = {
		name = "Convert all non-primary to labourers"
		
		any_pop = {
			limit = {
				NOT = { is_primary_culture = yes }
				NOT = { type = labourers }
				NOT = { type = slaves }
			}
			pop_type = labourers
		}
	}
	
	option = {
		name = "Go Back"
		
		country_event = 9500
	}
	
	option = {
		name = "Quit"
	}
}

# Slavery Options
country_event = {
	id = 9503
	title = "Choose a Slavery Option"
	desc = "EVTDESC9503"
	picture = administration
	
	is_triggered_only = yes
	
	option = {
		name = "Do not enslave conquered pops"
		
		clr_country_flag = poor_slaving
		clr_country_flag = all_slaving
	}
	
	option = {
		name = "Enslave poor conquered pops"
		
		set_country_flag = poor_slaving
		clr_country_flag = all_slaving
	}
	
	option = {
		name = "Enslave all non-accepted conquered pops"
		
		clr_country_flag = poor_slaving
		set_country_flag = all_slaving
	}
}

# Enslave poor
country_event = {
	id = 9504
	title = "Enslaving the New Poor"
	desc = "EVTDESC9504"
	picture = administration
	
	trigger = {
		any_owned_province = {
			any_pop = {
				is_accepted_culture = no
				is_primary_culture = no
				strata = poor
				NOT = { type = slaves }
			}
			is_slave = yes
		}
		NOT = { slavery = no_slavery }
		has_country_flag = poor_slavery
	}
	
	option = {
		name = "Our new slaves"
		
		any_pop = {
			limit = {
				is_accepted_culture = no
				is_primary_culture = no
				strata = poor
				location = { is_slave = yes }
			}
			pop_type = slaves
		}
	}
}

# Enslave all
country_event = {
	id  = 9505
	title = "Enslaving the Unaccepteds"
	desc = "EVTDESC9505"
	picture = administration
	
	trigger = {
		any_owned_province = {
			any_pop = {
				is_accepted_culture = no
				is_primary_culture = no
				NOT = { type = slaves }
			}
			is_slave = yes
		}
		NOT = { slavery = no_slavery }
		has_country_flag = all_slaving
	}
	
	option = {
		name = "Our new slaves"
		
		any_pop = {
			limit = {
				is_accepted_culture = no
				is_primary_culture = no
				location = { is_slave = yes }
			}
			pop_type = slaves
		}
		
		any_pop = [
			limit = {
				type = slaves
			}
			ideology = {
				value = slave_ideology
				factor = 10
			}
		}
	}
}
