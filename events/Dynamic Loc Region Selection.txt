# Dynamic Loc Region Selection 1
country_event = {
	id = 300008
	title = "Dynamic Loc Region Selection 1"
	desc = ""
	# picture = ""
	
	is_triggered_only = yes
	
	option = {
		name = "Assign Loc"
		
		clr_global_flag = dynamic_loc_option_1_found
		
		random_owned = {
			any_neighbor_province = {
				limit = {
					has_province_modifier = dynamic_loc_potential_region_target
					region = DAL_654
					NOT = { owner = { has_global_flag = dynamic_loc_option_1_found } }
				}
				railroading_tag_specific_description_loc_1 = { state_scope = { change_region_name = "Nan Annen" } }
				DAL_654 = { remove_province_modifier = dynamic_loc_potential_region_target }
				DAL_654 = {
					limit = {
						any_core = {
							NOT = { tag = FROM }
							has_country_modifier = alliance_of_dwarves_and_men
						}
					}
					add_province_modifier = { name = dynamic_loc_region_target_1 duration = -1 }
				}
				owner = { set_global_flag = dynamic_loc_option_1_found }
			}
		}
	}
}
