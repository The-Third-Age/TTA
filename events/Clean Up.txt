	id = 100011
	title = "Good Evil Relations Cleanup"
	desc = "EVTDESC100011"
	picture = "administration"
	is_triggered_only = yes
	
	option = {
		name = "AI, stop doing stupid alliances"
		
		any_neighbor_country = {
			limit = {
				good_evil_alignment = good_alignment
				ai = yes
			}
			
			country_event = 100012
		}
		
		any_neighbor_country = {
			limit = {
				good_evil_alignment = evil_alignment

			}

		}
		
		country_event = { id = 100011 days = 30 }
	}
}
