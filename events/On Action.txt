#$STATE$ Joins $COUNTRY$ - Slavery
country_event = {
	id = 70000
	title = "$STATE$ Granted Statehood!"
	desc = "EVTDESC70000"
	picture = "Slaves"
	
	is_triggered_only = yes
	allow_multiple_instances = yes
	
	trigger = {
		country = {
			OR = {
				civilized = no
				slavery = yes_slavery
			}
		}
	}
	
	option = {
		name = "Extend slavery"
		is_slave = yes
		country = {
			any_state = {
				limit = {
					is_slave = no
				}
				any_pop = { consciousness = 1 }
			}
			any_state = {
				limit = {
					is_slave = yes
				}
				any_pop = { consciousness = -0.5 }
			}
		}
		any_pop = {
			limit = {
				type = soldiers
				is_primary_culture = no
				is_accepted_culture = no
			}
			pop_type = tribals
		}
	}
}

#$STATE$ Joins $COUNTRY$! - Remove Cores
country_event = {
	id = 70001
	title = "$STATE$ Granted Statehood!"
	desc = "EVTDESC70001"
	
	is_triggered_only = yes
	allow_multiple_instances = yes
	
	trigger = {
		OR = {
			country = {
				tag = GON
				any_owned_province = {
					OR = {
						region = GON_160
						region = GON_165
						region = GON_169
						region = GON_175
						region = GON_183
					}
					OR = {
						is_core = PUK
						is_core = DRA
						is_core = DRH
					}
					is_colonial = no
				}
				has_country_flag = andrast_gondorification
			}
			country = {
				tag = GON
				any_owned_province = {
					is_core = HRD
					OR = {
						is_core = HRU
						is_core = HRN
						is_core = HAR
					}
					is_colonial = no
				}
				has_country_flag = south_gondor_gondorification
			}
		}
	}
	
	option = {
		name = "Excellent!"
		
		# Outcome of Gondorification of Andrast
		random_owned = {
			limit = {
				owner = {
					tag = GON
					has_country_flag = andrast_gondorification
					any_owned_province = {
						OR = {
							region = GON_160
							region = GON_165
							region = GON_169
							region = GON_175
							region = GON_183
						}
						OR = {
							is_core = PUK
							is_core = DRA
							is_core = DRH
						}
						is_colonial = no
					}
				}
			}
			owner = {
				any_owned = {
					limit = {
						OR = {
							region = GON_160
							region = GON_165
							region = GON_169
							region = GON_175
							region = GON_183
						}
						OR = {
							is_core = PUK
							is_core = DRA
							is_core = DRH
						}
						is_colonial = no
					}
					remove_core = PUK
					remove_core = DRA
					remove_core = DRH
				}
			}
		}
		
		# Outcome of Gondorification of Harondor
		random_owned = {
			limit = {
				owner = {
					tag = GON
					has_country_flag = south_gondor_gondorification
					any_owned_province = {
						OR = {
							is_core = HRU
							is_core = HAR
						}
						is_core = HRD
						is_colonial = no
					}
				}
			}
			owner = {
				any_owned = {
					limit = {
						is_core = HRD
						OR = {
							is_core = HRU
							is_core = HAR
						}
						is_colonial = no
					}
					remove_core = HRU
					remove_core = HAR
				}
			}
		}
		any_pop = {
			limit = {
				type = soldiers
				is_primary_culture = no
				is_accepted_culture = no
			}
			pop_type = tribals
		}
	}
}
