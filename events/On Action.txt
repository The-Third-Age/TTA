#$STATE$ Joins $COUNTRY$ - Slavery
country_event = {
	id = 70000
	title = "$STATE$ Granted Statehood!"
	desc = "EVTDESC70000"
	picture = "Slaves"
	trigger = {
		country = { slavery = yes_slavery }
	}
	is_triggered_only = yes
	allow_multiple_instances = yes
	
	option = {
		name = "Extend slavery"
		is_slave = yes
		country = {
			any_state = {
				limit = {
					is_slave = no
				}
				any_pop = { consciousness = 1 }
			}
			any_state = {
				limit = {
					is_slave = yes
				}
				any_pop = { consciousness = -0.5 }
			}
		}
	}
}

#$STATE$ Joins $COUNTRY$! - Remove Cores
country_event = {
	id = 70001
	title = "$STATE$ Granted Statehood!"
	desc = "EVTDESC70001"
	picture = "Western"
	trigger = {
		OR = {
			country = { 
				tag = GON
				any_owned_province = {
					OR = {
						region = GON_162
						region = GON_165
						region = GON_169
						region = GON_175
						region = GON_183
					}
					is_core = PUK
					is_colonial = no
				}
				has_country_flag = andrast_gondorification
			}
			country = {
				tag = GON
				any_owned_province = {
					is_core = HRD
					OR = {
						is_core = HRU
						is_core = HRN
						is_core = HAR
					}
					is_colonial = no
				}
				has_country_flag = south_gondor_gondorification
			}
		}
	}
	is_triggered_only = yes
	allow_multiple_instances = yes
	
	option = {
		name = "Excellent!"
		
		country = {
			any_owned = {
				limit = {
					owner = { tag = GON }
					OR = {
						region = GON_162
						region = GON_165
						region = GON_169
						region = GON_175
						region = GON_183
					}
					is_core = PUK
					is_colonial = no
				}
				remove_core = PUK
			}
		}
		
		country = {
			any_owned = {
				limit = {
					owner = { tag = GON }
					is_core = HRD
					OR = {
						is_core = HRU
						is_core = HRN
						is_core = HAR
					}
					is_colonial = no
				}
				remove_core = HRU
				remove_core = HRN
				remove_core = HAR
			}
		}
	}
}

country_event = {
	id = 70007
	title = "Freeing Our Brethren"
	desc = "EVTDESC70000"
	picture = "administration"
	
	trigger = {
		country = { slavery = non_accepted_slavery }
	}
	is_triggered_only = yes
	allow_multiple_instances = yes
	
	option = {
		name = "Extend slavery"
		is_slave = yes
		country = {
			any_state = {
				limit = {
					is_slave = no
				}
				any_pop = { consciousness = 1 }
			}
			any_state = {
				limit = {
					is_slave = yes
				}
				any_pop = { consciousness = -0.5 }
			}
			any_pop = {
				limit = {
					type = slaves
					OR = {
						is_primary_culture = yes
						is_accepted_culture = yes
					}
				}
				pop_type = labourers
			}
		}
	}
}
