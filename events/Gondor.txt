# The Second Kin-Strife
country_event = {
	id = 50017
	title = "The Council of Gondor"
	desc = "EVTDESC50017"
	# picture = ?
	is_triggered_only = yes
	
	option = {
		name = "A more fitting name is the Council of Traitors"
		
		# TODO: Make the rebel country
		
		any_owned = {
			limit = {
				is_core = THIS
				is_state_capital = yes
			}
			random = {
				chance = 50
				add_province_modifier = { name = province_selection duration = -1 }
			}
		}
		
		any_owned = {
			limit = {
				# TODO: Check to see if the state capital has the province_selection modifier
				# But not the state capital itself
			}
			secede_province = MOR
		}
		
		any_owned = {
			limit = {
				has_province_modifier = province_selection
			}
			secede_province = MOR
		}
		
		any_owned = {
			limit = {
				NOT = { is_core = THIS }
			}
			
			secede_province = MOR
		}
		
		any_country = {
			limit = {
				exists = yes
				substate_of = THIS
			}
			
			annex_to = MOR
		}
	}
}
