	mean_time_to_happen = {
		months = 60
		
		modifier = {
			factor = 0.9
			11 = { average_consciousness = 5 }

			factor = 0.9
			11 = { average_consciousness = 8 }
		}
		modifier = {
			factor = 0.9
			11 = { average_militancy = 7 }
		}
		modifier = {
			factor = 0.9
			11 = { average_militancy = 8 }
		}
		modifier = {
			factor = 0.9
			11 = { average_militancy = 9 }
		}
		modifier = {
			factor = 0.5
			11 = { average_militancy = 10 }
		}
	}
	
	option = {
		name = "They wouldn't!"
		
		prestige = -3
		
		any_pop = {
			limit = { 
				type = labourers 
				has_pop_culture = druedain

			}
			pop_type = soldiers
		}
		11 = { secede_province = DRU }
		any_pop = {
			limit = {
				has_pop_culture = druedain
				NOT = { location = { province_id = 11 } }
			}
			militancy = 1
			consciousness = 4
		}
		DRU = { country_event = { id = 50014 days = 90 } }
	}
}

			sub_unit = { type = archers value = current }
			sub_unit = { type = archers value = current }
			sub_unit = { type = archers value = current }
			sub_unit = { type = archers value = current }

			sub_unit = { type = pikemen value = current }
			sub_unit = { type = pikemen value = current }
			sub_unit = { type = pikemen value = current }
			sub_unit = { type = scouts value = current }
			sub_unit = { type = artillery value = current }
		}
		
		treasury = 100000
			
		add_casus_belli = {
			target = FROM
			type = conquest
			months = 12
		}
		FROM = {
			war = {
				target = DRU

				call_ally = yes
			}

		}
		set_global_flag = DRU_rebellion
	}
