			}
			add_core = GNR
		}

		any_country = {
			limit = {
				exists = yes
				OR = {
					substate_of = THIS

					}
				}
			}
			annex_to = GNR
		}
		
		all_core = { add_core = GNR }
		
		war = {
			target = GNR
			attacker_goal = { casus_belli = civil_war }
			defender_goal = { casus_belli = civil_war }
			call_ally = no
		}
	}
}
